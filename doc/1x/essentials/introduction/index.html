import { Page, PreparePage } from '@nimble-ts/core';

@PreparePage({ ... })
export class MyFirstPage extends Page {

    private timer: any;
    public seconds: number = 0;

    constructor() {
        super();
    }

    onInit() {
        this.timer = setInterval(() => {
            this.seconds += 1;
            this.render();
        }, 1000);
    }

    onDestroy() {
        clearInterval(this.timer);
    }

}

     {{seconds}}

import { Page, PreparePage } from '@nimble-ts/core';

@PreparePage({ ... })
export class MyFirstPage extends Page {

    public show: boolean = true;

    constructor() {
        super();
    }

    public toggleMsg() {
        this.render(() => {
            this.show = !this.show;
        });
    }

}
