import { PrepareDirective, Directive } from '@nimble-ts/core/directives';

@PrepareDirective({
    selector: 'force-focus'
})
export class ForceFocusDirective extends Directive {

	constructor() {
		super();
	}

    public onResolve(selector: string, value: any): void {
		this.element.focus();
	}

    public onDestroy() {
    }
}
